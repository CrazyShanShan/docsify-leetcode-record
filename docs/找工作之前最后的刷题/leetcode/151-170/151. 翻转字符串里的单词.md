## [151. 翻转字符串里的单词](https://leetcode-cn.com/problems/reverse-words-in-a-string/)

**analyse:**

```
1. 去除空格
2. 反转整个链表
3. 反转每个单词
```

**code:**

1. 自己实现的

```java
class Solution {
    public String reverseWords(String s) {
        StringBuilder sb = trim(s);
        sb = reverse(sb, 0, sb.length() - 1);
        sb = reverse(sb);
        return sb.toString();
    }
    
    private StringBuilder trim(String s) {
        StringBuilder sb = new StringBuilder();
        int n = s.length();
        int x = 0;
        while(x < n) {
            while (x < n && s.charAt(x) == ' ') {
                x++;
            }
            while (x < n && s.charAt(x) != ' ') {
                sb.append(s.charAt(x));
                x++;
            }
            if (x < n) sb.append(' ');
        }
        if (sb.charAt(sb.length() - 1) == ' ') sb.deleteCharAt(sb.length() - 1);
        return sb;
    }
    
    private StringBuilder reverse(StringBuilder sb, int st, int ed) {
        for (int i = st, j = ed; i <= j; i++, j--) {
            char ch = sb.charAt(i);
            sb.setCharAt(i, sb.charAt(j));
            sb.setCharAt(j, ch);
        }
        return sb;
    }
    
    private StringBuilder reverse(StringBuilder sb) {
        int n = sb.length();
        int x = 0;
        while (x < n) {
            int y = x + 1;
            while (y < n && sb.charAt(y) != ' ') y++;
            reverse(sb, x, y - 1);
            x = y + 1;
        }
        return sb;
    }
    
    
}
```

2. y总思路

```java
class Solution {
    public String reverseWords(String s) {
        StringBuilder sb = new StringBuilder(s);
        int k = 0;
        for (int i = 0; i < sb.length(); i++) {
            if (sb.charAt(i) == ' ') continue;
            int j = i, t = k;
            while (j < sb.length() && s.charAt(j) != ' ') sb.setCharAt(k++, sb.charAt(j++));
            reverse(sb, t, k - 1);
            if (k < sb.length()) sb.setCharAt(k++, ' ');
            i = j;
        }
        if (sb.charAt(k - 1) == ' ')k -= 1;
        sb.delete(k, sb.length());
        reverse(sb, 0, k - 1);
        return sb.toString();
    }
    
    private void reverse(StringBuilder sb, int st, int ed) {
        for (int i = st, j = ed; i <= j; i++, j--) {
            char ch = sb.charAt(i);
            sb.setCharAt(i, sb.charAt(j));
            sb.setCharAt(j, ch);
        }
    }
}
```
